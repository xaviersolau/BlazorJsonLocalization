using Microsoft.Extensions.Localization;
using SoloX.BlazorJsonLocalization;
using SoloX.BlazorJsonLocalization.Tools.Core.Samples.SampleMethodArg;
using System.Collections.Generic;
#pragma warning restore IDE0005 // Using directive is unnecessary.

namespace SoloX.BlazorJsonLocalization.Tools.Core.Samples.SampleMethodArg
{
    /// <summary>
    /// ISimpleLocalizer generated implementation.
    /// </summary>
    public partial class SimpleLocalizer : ISimpleLocalizer
    {
        private readonly IStringLocalizer<Component> stringLocalizer;

        /// <summary>
        /// Setup instance.
        /// </summary>
        /// <param name="stringLocalizer">Base string localizer.</param>
        public SimpleLocalizer(IStringLocalizer<Component> stringLocalizer)
        {
            this.stringLocalizer = stringLocalizer;
        }

        /// <inheritdoc/>
        public LocalizedString this[string name]
            => this.stringLocalizer[name];

        /// <inheritdoc/>
        public LocalizedString this[string name, params object[] arguments]
            => this.stringLocalizer[name, arguments];

        /// <inheritdoc/>
        public IEnumerable<LocalizedString> GetAllStrings(bool includeParentCultures)
            => this.stringLocalizer.GetAllStrings(includeParentCultures);

        /// <summary>
        /// Get MethodWithArgString localized string.
        /// </summary>
        public string MethodWithArgString(string arg)
            => this.stringLocalizer[nameof(MethodWithArgString), arg];

        /// <summary>
        /// Get MethodWithArgInt localized string.
        /// </summary>
        public string MethodWithArgInt(int arg)
            => this.stringLocalizer[nameof(MethodWithArgInt), arg];

        /// <summary>
        /// Get MethodWithArgDouble localized string.
        /// </summary>
        public string MethodWithArgDouble(double arg)
            => this.stringLocalizer[nameof(MethodWithArgDouble), arg];

        /// <summary>
        /// Get MethodWithArgDateTime localized string.
        /// </summary>
        public string MethodWithArgDateTime(DateTime arg)
            => this.stringLocalizer[nameof(MethodWithArgDateTime), arg];

        /// <summary>
        /// Get MethodWithArgDateOnly localized string.
        /// </summary>
        public string MethodWithArgDateOnly(DateOnly arg)
            => this.stringLocalizer[nameof(MethodWithArgDateOnly), arg];
    }


}

        /// Convert StringLocalizer to ISimpleLocalizer

            => new SimpleLocalizer(stringLocalizer);

        /// Get MethodWithArgString localized value from the String Localizer.
